\begin{lemma}
\label{lem:min-eig} 
Consider a network separable family $\{\sepmodel : \nat \in \mbR^p\}$ satisfying \eqref{general model} 
and an observation $\bx \in \mbX$ of $\bX$ 
and let $(\bx, \by)$ be the network concordant pair where $\by$ is given by Proposition \ref{prop:inference}. 
Define,
for each pair of nodes $\{i,j\} \subset \mN$,
\beno
L_{i,j}(\nat, \bx_{i,j}, \by)
&\coloneqq& \log \, \sepmodel(\bX_{i,j} = \bx_{i,j} \,|\, \bY = \by) \s \\ 
\widetilde{L}_{i,j}(\nat, \bx_{i,j}, \by)
&\coloneqq& \dsum_{k=1}^{K} \log \, \sepmodel(X_{i,j}^{(k)} = x_{i,j}^{(k)} \,|\, \bX_{i,j}^{(-k)} = \bx_{i,j}^{(-k)}, \bY = \by).
\ee
Then there exist $p \times p$ matrices 
$\mcI(\nat)$ and $\widetilde\mcI(\nat)$ such that 
%$\mcI_{i,j}(\nat)$ and  $\widetilde\mcI_{i,j}(\nat)$
\beno
\mbE\left[ - \nabla_{\nat}^2 L_{i,j}(\nat, \bX_{i,j}, \bY) \,|\, \bY = \by \right] 
\=
\begin{cases}
    \mcI(\nat) & Y_{i,j} = 1 \\
    \bm{0}_{p,p} & Y_{i,j} = 0
\end{cases} \s \\
\mbE\left[ - \nabla_{\nat}^2 \widetilde{L}_{i,j}(\nat, \bX_{i,j}, \bY) \,|\, \bY = \by \right] 
\= 
\begin{cases}
    \widetilde\mcI(\nat) & Y_{i,j} = 1 \\
    \bm{0}_{p,p} & Y_{i,j} = 0,
\end{cases} 
\ee
for all $\{i,j\} \subset \mN$,  
where $\bm{0}_{p,p}$ is the $p\times p$ matrix with all $0$ entries, 
and 
\beno 
&& \lambda_{\min}(-\mbE \, \nabla_{\nat}^2 \, \ell(\nat; \bX, \bY)) 
\= \lambda_{\min}(\mcI(\nat)) \, \mbE \, \norm{\bY}_1 \s \\ 
&& \lambda_{\min}(-\mbE \, \nabla_{\nat}^2 \, \pl(\nat; \bX, \bY)) 
\= \lambda_{\min}(\widetilde\mcI(\nat)) \, \mbE \norm{\bY}_1,  
\ee
where $\lambda_{\min}(\bA)$ is the smallest eigenvalue of matrix $\bA \in \mbR^{p \times p}$.
\end{lemma}
