\section{Macroscopic logs from Sen}

\subsection{$T^4$}
We consider IIB supergravity reduced on $T^4$ and then reduce on $S^1$. We shall ignore scalars and fermions, and only keep differential forms, gravitinos, and the graviton, as they can potentially have zero modes. The relevant 5d field content is summarized in the following table.
\begin{center}
\begin{tabular}{ |c|c|c|c|c| } 
 \hline
 10d & $g_{\mu\nu}$ & $B_{\mu\nu}$ & $C_{\mu\nu}$ & $C_{\alpha\beta\gamma\delta}^+$\\
 \hline\hline
5d & $g_{\mu\nu}$,$5A_\mu$ & $5B_{\mu}$,$B_{\mu\nu}$ & $5C_{\mu}, C_{\mu\nu}$ &$(6+4)\times C_\alpha$ \\ 
 \hline
\end{tabular}
\end{center}


% In the above table the number 10 for $C_\alpha$ (or the dual $C_{\alpha\beta}$) comes from $b_2$ of the Enriques surface being 10, as seen from .

 We obtain the coefficient of $\log\Lambda^{1/2}$ in the logarithmic correction to be
\begin{equation}
\begin{split}
&\underbrace{-(\frac{3}{2}-1)\times 27}_{\text{gauge fields}} -\underbrace{(\frac{5}{2}-1)\times (3+4)}_{\text{metric}}
+\underbrace{(4-1)\times 4}_{\text{gravitino}}=-12,\\
\end{split}\label{eq:logcorrection5d}
\end{equation}

\subsection{$\mathrm{HS}$}
We consider IIB supergravity reduced on $T^4$ (as the $T^4$ zero-modes are trivially invariant under the $\mathbb{Z}_2$) and then reduce on $S^1$. We shall ignore scalars and fermions, and only keep differential forms, gravitinos, and the graviton, as they can potentially have zero modes. The relevant 5d field content is summarized in the following table.
\begin{center}
\begin{tabular}{ |c|c|c|c|c| } 
 \hline
 10d & $g_{\mu\nu}$ & $B_{\mu\nu}$ & $C_{\mu\nu}$ & $C_{\alpha\beta\gamma\delta}^+$\\
 \hline\hline
5d & $g_{\mu\nu}$,$3A_\mu$ & $3B_{\mu}$,$B_{\mu\nu}$ & $3C_{\mu}, C_{\mu\nu}$ &$(2+2)\times C_\alpha$ \\ 
 \hline
\end{tabular}
\end{center}

We obtain the coefficient of $\log\Lambda^{1/2}$ in the logarithmic correction to be
\begin{equation}
\begin{split}
&\underbrace{-(\frac{3}{2}-1)\times 15}_{\text{gauge fields}} -\underbrace{(\frac{5}{2}-1)\times (3+2)}_{\text{metric}}
+\underbrace{(4-1)\times 2}_{\text{gravitino}}=-9,\\
\end{split}\label{eq:logcorrection5d}
\end{equation}
