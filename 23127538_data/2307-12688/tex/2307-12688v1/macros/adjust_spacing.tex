% ~ sections {command}{left}{before}{after}[right]

% ~ restatable without post space
\makeatletter
\let\oldendrestatable\endthmt@restatable
\renewcommand\endthmt@restatable{\oldendrestatable\!}%\vspace{-2ex}
\makeatother

% ~ restatable
\makeatletter
\def\restatable@space@setup{%
  \restatable@preskip=1ex plus0.5ex minus0.5ex%
  \restatable@postskip=2ex plus1ex minus1ex%
}
\makeatother
