
% ~ creates recursive definition
% #1 : s* : math mode wrap
% #2 : o  : rec label identifier
% #3 : E: : default recursive label
% #4 : e. : proceeding type
\NewDocumentCommand{\IRecDef}{s o E:{\TypRecLabel} e.}{%
    \IfBooleanT{#1}{$}% mm wrap
    {{#3}\IfValueT{#2}{^{#2}}}%
    % * use proceeding type if given, else end
    \IfValueTF{#4}{.\,{#4}}{.\TypeEnd}%
    \IfBooleanT{#1}{$}% mm wrap
}

% ~ creates recursive call
% #1 : s* : math mode wrap
% #2 : o  : label identifier
% #3 : E:  : default recursive label
\NewDocumentCommand{\IRecCall}{s o E:{\TypRecLabel}}{%
    \IfBooleanT{#1}{$}% mm wrap
    {{#3}\IfValueT{#2}{^{#2}}}%
    \IfBooleanT{#1}{$}% mm wrap
}
