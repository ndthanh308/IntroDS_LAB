
% ~ smarter par
\NewDocumentCommand{\Parl}{s m}{%
    \IfBooleanT{#1}{$}%
    \ListToParl{#2}%
    \IfBooleanT{#1}{$}%
}


% ~ tex math mode spacing
\NewDocumentCommand\ListToParl{>{\SplitList{,}}m}
{%
    {\ProcessList{#1}{\HeadParlContents}}%
}

% ~ 
\newcommand\HeadParlContents[1]{%
    {#1}%
    \let\HeadParlContents\TailParlContents%
}

% ~ inserts | before each subsequent par item
\newcommand\TailParlContents[1]{%
    \!\,{\mid}\;\!{#1}%
}
