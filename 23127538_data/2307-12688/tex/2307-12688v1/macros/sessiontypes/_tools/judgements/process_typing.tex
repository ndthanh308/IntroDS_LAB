
\NewDocumentCommand{\IPJEnv}{s O{\VarEnv} e:}{%
    \IfBooleanT{#1}{$}%
    % * Gamma
    {#2}%
    \IfValueT{#3}{% * Gamma contents?
        \ListToEnvItems{#3}%
    }%
    \IfBooleanT{#1}{$}%
}

\NewDocumentCommand{\IPJPrc}{s o}{%
    \IfBooleanT{#1}{$}%
    % * default process
    \IfNoValueTF{#2}{\Prc}{% * else list of par processes
        \Parl{#2}%
    }%
    \IfBooleanT{#1}{$}%
}

\NewDocumentCommand{\IPJSes}{s O{\VarEnv} e:}{%
    \IfBooleanT{#1}{$}%
    % * Delta
    {#2}%
    \IfValueT{#3}{% * Delta contents?
        \ListToEnvItems{#3}%
    }%
    \IfBooleanT{#1}{$}%
}

\NewDocumentCommand{\IPJudgement}{s O{\VarEnv} e: t< O{\Prc} t> O{\SesEnv} e:}{%
    \IfBooleanTF{#1}{$\FmtJudgement}{\FmtJudgement|}%
    {% * Gamma
        \IfNoValueTF{#3}{\IPJEnv[#2]}{% * Gamma vars
            \IPJEnv[#2]:{#3}%
        }%
    }{% * Process
        \IPJPrc[#5]%
    }[% * Delta
        \IfNoValueTF{#8}{\IPJSes[#7]}{% * Delta roles
            \IPJSes[#7]:{#8}%
        }%
    ]%
    \IfBooleanT{#1}{$}%
}
