
\NewDocumentCommand{\RuleLbl}{m m o}{%
   \IfNoValueTF{#3}{%
      \IfBooleanTF{#1}{%
         \RName*[#2]%
      }{%
         \RName[#2]%
      }%
   }{%
      \IfBooleanTF{#1}{%
         \RName*[#2][#3]%
      }{%
         \RName[#2][#3]%
      }%
   }%
}
