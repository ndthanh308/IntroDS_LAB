
% ~ 
% #1 : s* : math mode wrap
% #2 : E: : process variable
% #3 : o  : recursion label script
% #4 : E= : recursion label
% #5 : e. : proceeding process
\NewDocumentCommand{\PDefRec}{s E:{\RecPrcVar} o E={\TypRecLabel} e.}{%
    \IfBooleanT{#1}{$}%
    \begin{array}[t]{l l}%
        \TTPrcDef & %
        \IfValueTF{#3}{{#2}^{#3}={#4}^{#3}}{{#2}={#4}}\;%
        \TTPrcDefIn:%
        \\ & %
        \rule{0pt}{1ex}%
        % \hspace{6ex}%
        \IfValueTF{#5}{% * proceeding process
            {#5}
        }{\PrcEnd}%
        \;%
    \end{array}%
    \IfBooleanT{#1}{$}%
}

% ~ 
% #1 : s* : math mode wrap
% #2 : E: : recursion label
% #3 : o : recursion label script
\NewDocumentCommand{\PRec}{s E:{\TypRecLabel} o}{%
    \IfBooleanT{#1}{$}%
    \IfValueTF{#3}{{#2}^{#3}}{{#2}}%
    \IfBooleanT{#1}{$}%
}
