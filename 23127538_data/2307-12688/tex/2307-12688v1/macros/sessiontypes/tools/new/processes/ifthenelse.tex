
% ~ 
\NewDocumentCommand{\PIf}{s O{true} E:{\PThen} E:{\PElse}}{%
    \IfBooleanT{#1}{$}%
    \begin{array}[t]{l l}%
        \TTPrcIf\;\mathrlap{\mathtt{(#2)}}%
        % \hspace{\widthof{{(#2)}}}%
        \\ #3%
        \\ #4%
    \end{array}%
    \IfBooleanT{#1}{$}%
}

% ~ 
\NewDocumentCommand{\PThen}{s O{\PrcEnd}}{%
    \IfBooleanT{#1}{$}%
    % \mathllap{\TTPrcThen} & : & \begin{array}[t]{l}{#2}\end{array}%
    {\TTPrcThen} & {: \begin{array}[t]{l}{{#2}}\end{array}}%
    \IfBooleanT{#1}{$}%
}

% ~ 
\NewDocumentCommand{\PElse}{s O{\PFail}}{%
    \IfBooleanT{#1}{$}%
    % \mathllap{\TTPrcElse} & : & \begin{array}[t]{l}{#2}\end{array}%
    {\TTPrcElse} & {: \begin{array}[t]{l}{{#2}}\end{array}}%
    \IfBooleanT{#1}{$}%
}
