
% ~ 
\NewDocumentCommand{\IScope}{s O{\ISes} e.}{%
    \IfBooleanT{#1}{$}% mm wrap
    \begin{array}[t]{l l}%
        \begin{array}[t]({l})%
            {\PrcScope}{#2}%
        \end{array}%
        & %
        % * use proceeding type if given, else end
        \begin{array}[t]({l})%
            \IfValueTF{#3}{#3}{\PrcEnd}%
        \end{array}%
    \end{array}%
    \IfBooleanT{#1}{$}% mm wrap
}

% ~ 
\NewDocumentCommand{\PDelay}{s O{0} t/ e.}{%
    \IfBooleanT{#1}{$}%
    \begin{array}[t]{l l}%
        \IfBooleanTF{#3}{% * new line proceeding
            \mathrlap{\FuncDelay\left({#2}\right)\;\!.} & %
            \\ \mbox{ } & {#4}%
        }{% * same line
            \FuncDelay\left({#2}\right)\;\!.\,%
            {#4}%
        }%
    \end{array}%
    \IfBooleanT{#1}{$}%
}

% ~ 
\NewDocumentCommand{\PSet}{s m t/ e.}{%
    \IfBooleanT{#1}{$}%
    \begin{array}[t]{l}%
        \IfBooleanTF{#3}{% * new line proceeding
        \mathrlap{\TTPrcSet\left(\mathtt{#2}\right)} %
        \\ .{#4}%
        }{% * same line
        \TTPrcSet\left(\mathtt{#2}\right).{#4}%
        }%
    \end{array}%
    \IfBooleanT{#1}{$}%
}

% ~ 
\NewDocumentCommand{\PFail}{s}{%
    \IfBooleanT{#1}{$}%
    \PrcFail%
    \IfBooleanT{#1}{$}%
}
