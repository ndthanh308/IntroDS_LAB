\begin{minieq}\label{eq:types_predicates}
    \begin{array}[c]{>{$} p{0.65\textwidth} <{$} r}
        \Const ::= \TypeTrue%
        \mid \Clx>\ValTime%
        \mid \Clx=\ValTime%
        \mid \Clx-\Cly>\ValTime%
        \mid \Clx-\Cly=\ValTime%
        \mid \neg\Const%
        \mid \Const_1\land\Const_2%
        %
            & \text{where\ }\Clx,\Cly\in\ClockSet %
    \end{array}
\end{minieq}
