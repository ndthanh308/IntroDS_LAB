\begin{definition}[Type Progress]\label{def:types_progress}
    %
    A configuration with queues $\SocCfg$ is \emph{final} if $\VSoc=\CSoc;[\TypeEnd]:{\emptyset}$.
    A system \VSys*\ \emph{satisfies progress} for all \VSys*'\ reachable from \VSys*, either \VSoc*'_1\ and \VSoc*'_2\ are \emph{final}, or $\exists\ValTime$ such that \Trans*{\VSys'}:{\ValTime,\SiltAction}.
  \end{definition}
