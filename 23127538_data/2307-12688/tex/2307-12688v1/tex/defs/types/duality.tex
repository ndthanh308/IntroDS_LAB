\begin{definition}[Dual Types]\label{def:types_dual}
   %
   The dual type \Dual*\ of type \TypeS*\ is defined as follows:
   \begin{minieq}*
      \begin{array}[t]{r c l p{7ex} r c l p{7ex} r c l p{7ex} r c l}
         %
         \Dual[\TypRecDef]
         & =
         & \TypRecDef.\Dual[\TypeS]
         &
         &                                                              %
         \Dual[\TypRecLabel]
         & =
         & \TypRecLabel
         &
         &                                                              %
         \Dual[\TypeEnd]
         & =
         & \TypeEnd
         &
         &                                                              %
         \Dual[\TypSend]
         & =
         & \TypRecv
         %
         \\[1ex]
         &
         &
         &
         &                                                              %
         \mathllap{\TypInteract~}
         & =
         & \mathrlap{\TypInteract1{1{\Dual[\TypComm]}}3{\Dual[\TypeS]}}
         &
         &
         &
         &
         &
         &                                                              %
         \Dual[\TypRecv]
         & =
         & \TypSend
         %
      \end{array}
   \end{minieq}
   \vspace{-2ex}
   %
   % \end{restatable}
\end{definition}
