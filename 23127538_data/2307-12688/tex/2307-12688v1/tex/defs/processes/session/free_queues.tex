\begin{definition}[Free Queues of \SesEnv*]\label{def:session_free_queues}
    %
    The function \PFuncFQ* is defined inductively as follows:
    \begin{minieq}
        \begin{array}[t]{c p{0.1\textwidth} c}
            \begin{array}[t]{r c l}%
                %
                \PFuncFQ[\emptyset]                        & = & \emptyset %
                %
                \\[1ex]%
                \PFuncFQ[\PrcP:\mkTup[\ValClocks][\TypeS]] & = & \emptyset %
            \end{array}
            &  & %
            \begin{array}[t]{r c l}%
                %
                \PFuncFQ[\PrcP\PrcQ:\Queue]   & = & \PrcP\PrcQ                                 %
                %
                \\[1ex]%
                \PFuncFQ[\SesEnv_1,\SesEnv_2] & = & \PFuncFQ[\SesEnv_1]\cup\PFuncFQ[\SesEnv_2] %
                %
            \end{array}
        \end{array}
    \end{minieq}
    %
\end{definition}
