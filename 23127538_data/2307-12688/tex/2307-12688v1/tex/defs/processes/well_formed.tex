\begin{definition}[Well-formed Processes]\label{def:prc_well_formed}
    %
    The \emph{well-formedness} of a process \Prc*\ is given by \PFuncWF*[\Prc], which is inductively defined as follows:
    \begin{minieq}*%
        \begin{array}[t]{l}%
            \begin{array}[c]{l p{8ex} l}%
                \begin{array}[c]{l c p{0ex} l}%
                    \begin{array}[t]{l}%
                        \PFuncWF[\RecVar], %
                        \mbox{}~\PFuncWF[\PCfgQueue],%
                        \mbox{}~\PFuncWF[\PrcEnd]%
                    \end{array}%
                     & = &  & %
                    \TypeTrue %
                \end{array}%
                %
                 &  & %
                \begin{array}[c]{l c p{0ex} l}%
                    \PFuncWF[\PCalcScope]%
                     & = &  & %
                    \PFuncWF[\Prc] %
                \end{array}%
            \end{array}%
            %
            \\[-1ex]\\%
            \begin{array}[c]{l c p{0ex} l}%
                \PFuncWF[\PCalcRecv] %
                 & = &  & %
                \bigwedge\limits_{i\in\SetI}~\PFuncWF[\Prc_i]%
                \land%
                \left(\PFuncFQ[\Prc_i]=\emptyset\right)%
            \end{array}%
            %
            \\[-1ex]\\%
            \begin{array}[c]{l c p{0ex} l}%
                \begin{array}[t]{l}%
                    \PFuncWF[\PCalcIf], %
                    \mbox{}~\PFuncWF[\Par[\Prc][\Qrc]]%
                \end{array}%
                 & = &  & %
                \PFuncWF[\Prc]~\land~\PFuncWF[\Qrc]%
            \end{array}%
            %
            \\[-1ex]\\%
            \begin{array}[c]{l c p{0ex} l}%
                \begin{array}[t]{l}%
                    \PFuncWF[\PCalcSend],      %
                    \mbox{}~\PFuncWF[\PCalcSetTimer],  %
                    \mbox{}~\PFuncWF[\PFuncDelay[\Const]], %
                    \mbox{}~\PFuncWF[\PFuncDelay[\ValTime]], \\%
                    \mbox{}~\PFuncWF[\PrcRecDef]             %
                \end{array}%
                 & = &  & %
                \PFuncWF[\Prc]%
                \land%
                \left(\PFuncFQ[\Prc]=\emptyset\right) %
            \end{array}%
            %
        \end{array}%
    \end{minieq}%
    %
\end{definition}
