    \begin{definition}[System Progress]\label{def:configs_sys_progress}
    %
    A system \VSys*\ \emph{satisfies progress} if:
    \begin{minieq}*
        \Trans{\VSys}:{\VSys'}*
        %
        \implies
        \begin{array}[c]{l}
            %
            \VSoc'_1
            \;\text{and\ }
            \VSoc'_2
            \;\text{are both a \emph{success} (see~\cref{def:types_progress})};
            %
            \\[0ex]
            \text{or\ } \exists\ValTime:
            \Trans{\VSys'}:{\ValTime\,\SiltAction}
            %
        \end{array}
        %
    \end{minieq}
    %
% \end{restatable}
\end{definition}
