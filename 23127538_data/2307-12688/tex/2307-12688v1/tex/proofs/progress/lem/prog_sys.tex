\begin{lemma}\label{lem:prog_sys}
   \REDO[The following holds:]
   \begin{equation*}\begin{array}{l}
         \begin{array}{l}
            \forall\mst,\mval:\quad%
            \emptyset;\,\mconstr\imp\mst%
            \quad\land\quad%
            \mval\models\mconstr:%
         \end{array} \\\qquad\qquad%
         \begin{array}{r}
            (\of{\mval}{0},\,\mst,\emptyset)%
            \,\mid\,%
            (\of{\mval}{0},\,\mdual[\mst],\emptyset)%
            \quad\text{satisfies progress (\cref{def:type_progress})}
         \end{array}
      \end{array}
   \end{equation*}
\end{lemma}
\begin{proof}
   Combination of~\cref{lem:prog_trans_iso_wf,lem:prog_trans_soc_compat,lem:prog_trans_compat_wf}.
\end{proof}
