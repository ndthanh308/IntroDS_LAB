\begin{lemma}\label{lem:prog_trans_compat_wf}
   Given two participants that are compatible \compconfig{}{1}{}{2}, and, both \isoconfig{1}\ and \isoconfig{2} are \emph{well-formed}:\par
   %
   If \;$\mparconfig{}{1}{}{2} {\transition{}}^{*} \mparconfig{'}{1}{'}{2}$,\par\noindent
   \hspace{7ex}\rule{0pt}{2.5ex} then \compconfig{'}{1}{'}{2},\par\noindent
   \hspace{9ex}\rule{0pt}{2.5ex}and, both \isoconfig[']{1}\ and \isoconfig[']{2} are \emph{well-formed}.
   %
\end{lemma}
\begin{proof}
   \REDO By~\cref{lem:prog_trans_iso_wf,lem:prog_trans_soc_compat}, it is shown that any transition taken by a system composed of a configurations \soc{1}\ and \soc{2}\ running in parallel, preserves the properties of \emph{well-formedness} and \emph{compatibility}.
\end{proof}
