\section{Proof of \cref{thm:main}}

By \cref{lem:almost-monochrom-rects}, 
we know any submatrix $\ms$ of $\mm$ contains a $(1-\frac{1}{16r})$-monochromatic rectangle of size
$\exp(-\Delta^{O(\Delta)} \cdot \sqrt{r} \cdot \log r) |\ms|$.
Therefore, by \cref{lem:almost-monochrom-rec-reduction}, $\ms$ contains a monochromatic rectangle of size
$\frac{1}{8} \exp(-\Delta^{O(\Delta)} \cdot \sqrt{r} \cdot \log r) |\ms|$.
We substitute this value for $\delta$ in \cref{lem:comm-protocol} to conclude
\begin{align*}
	CC^{\det}(\mm) &\leq \Theta \left( \log r \cdot \log \left(\frac{r}{\delta} \log(\Delta + 1) \right) \right) \\
	&= \Theta \left( \log^2 r + \log r \log \log (\Delta + 1) + \log^2 r \cdot \Delta^{O(\Delta)} \cdot \sqrt{r} \right) \\
	&= \Theta(\Delta^{O(\Delta)} \cdot \sqrt{r} \cdot \log^2 r).
\end{align*}
\qed
