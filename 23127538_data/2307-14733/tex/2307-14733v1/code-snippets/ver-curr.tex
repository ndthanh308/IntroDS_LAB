@Test
public void test_gt() throws Exception {
  CallableStatement cs =
    mock(CallableStatement.class);
  LocalDateTime LOCAL_DATE_TIME =
    LocalDateTime.now();
  TypeHandler<LocalDateTime> TYPE_HANDLER =
    new LocalDateTimeTypeHandler();
  // ======| \(S_{gt}\)|


  when(cs.getObject(1, LocalDateTime.class))
    .thenReturn(LOCAL_DATE_TIME);
  // ======
  assertEquals(
    LOCAL_DATE_TIME,
    TYPE_HANDLER.getResult(cs, 1)
  );
  verify(cs, never()).wasNull();
}
