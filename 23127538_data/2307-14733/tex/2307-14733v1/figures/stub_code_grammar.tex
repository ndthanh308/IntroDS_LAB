    \textit{VarDef} & \coloneqq v \leftarrow \textit{Expr} \\
    \textit{Expr} & \coloneqq \textit{Literal} \mid \textsf{Array}(v^*) \mid \textit{API}(v^*) \mid \textit{Mock} \\

    \textit{Reaction} & \coloneqq \textsf{Return}(v) \mid \textsf{Throw}(v)
\end{align*}
