\section{Facts about Euclidean mirror gradient flows}

Consider the Euclidean mirror gradient flow of a function $F:\rr^d\rightarrow \rr$ where the mirror $u:\rr^d \rightarrow \rr$ is a strictly convex function. Assume that, for some $c>0$, $\norm{\nabla^2 u}_\infty\le c$ and $\norm{(\nabla^2 u)^{-1}}_\infty \le c$. We show below that the mirror gradient flow is unique and converges exponentially fast to its equilibrium. 

Let $x_0, z_0$ denote the initial points of two mirror flows $(x_t,\; t\ge 0)$ and $(z_t,\; t\ge 0)$.

 

\begin{prop}
Let $F^*(y)=F(y^{u^*})$ denote the composition. 
Suppose that both $F$ and $F^*$ are $\lambda$-semiconvex, where $y^{u^*}=\left(\nabla u \right)^{-1}(y)$.  Then 
\[
\norm{x_t-z_t}^2 \le c(x_t - z_t) \cdot (x_t^u - z_t^u) \le ce^{-2\lambda c t} (x_0 - z_0) \cdot (x_0^u - z_0^u).
\]
\end{prop}

\begin{proof}
Fix some $z\in \rr^d$. Let 
\[
a_t = (x_t - z) \cdot (x^u_t - z^u), \quad t \ge 0.
\]
This is a nonnegative quantity since 
\[
a_t=(x_t - z) \cdot (\nabla u(x_t) - \nabla u(z))= \int_0^1 (x_t -z) \nabla^2 u(w_u) (x_t -z)du,
\]
where $w_u= (1-u)x_t + u z$. 

Then,
\[
\dot{a}_t = \dot{x}_t \cdot (x^u_t - z^u) + (x_t - z) \cdot \dot{x^u}_t= -\frac{\partial F}{\partial x^u}(x_t) \cdot (x^u_t - z^u) - \frac{\partial F}{\partial x}(x_t)\cdot (x_t - z),
\]
where the second equality follows from the definition of the mirror flow. 

By our assumed $\lambda$-semiconvexity,
\[
\begin{split}
F(z) - F(x_t) - \frac{\lambda}{2} \norm{x_t - z}^2 &\ge -\frac{\partial F}{\partial x}(x_t)\cdot (x_t - z)\\
F^*(z^u) - F^*(x^u_t) - \frac{\lambda}{2} \norm{x^u_t - z^u}^2 & \ge -\frac{\partial F^*}{\partial y}(x^u_t) \cdot (x^u_t - z^u)= -\frac{\partial F}{\partial x^u}(x_t) \cdot (x^u_t - z^u).
\end{split}
\]
Thus 
\[
\dot{a}_t\le F(z) - F(x_t) + F^*(z^u) - F^*(x^u_t) - \frac{\lambda}{2} \norm{x_t - z}^2 - \frac{\lambda}{2} \norm{x^u_t - z^u}^2.
\]

Now, by our assumption that $\norm{\nabla^2 u}_\infty \le c$, 
\[
a_t= (x_t - z)\cdot (\nabla u(x_t) - \nabla u(z))\le c \norm{x_t-z}^2.
\]
Similarly, $a_t \le c \norm{x_t^u - z^u}^2$ since $\norm{\nabla^2 u^*}_\infty \le c$. Hence, 
\[
\dot{a}_t \le F(z) - F(x_t) + F^*(z^u) - F^*(x^u_t) - c \lambda a(t). 
\]

Now, let 
\[
\xi_t= (x_t - z_t) \cdot (x_t^u - z_t^u).
\]
Repeating the above argument once for $x$ and again for $z$ and adding the two, we get 
\[
\dot{\xi}_t\le -2c\lambda \xi_t. 
\]
Thus $\xi_t \le e^{-2c\lambda t}$, $t\ge 0$, completing the proof.
\end{proof}
