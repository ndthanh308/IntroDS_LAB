\subsection{One-Step}
For a time-step size $\step > 0$, we a discrete approximation of continuous dynamics
\begin{align}
x_{t+1} = x_t + \step f(x_{t},u_t)
\end{align}
Throughout, we fix a horizon $H$ and a set of initial conditions $\cX_0$. For a given $B > 0$, we define the set $\cU_{\ell}(B) := \{u_{1:\ell} \in \cU^{\ell}: \max_{i \in [\ell]}\|u_i\| \le B\}$, with $\cU_{\ell} = \cU_{\ell}(\infty)$. For $u_{1:\ell} \in \cU_{\ell}$ We define $x_{\ell}(u_{1:\ell}\mid \xi)$ as the state $x_\ell$, where $x_1 = \xi$ and $x_{i+1} = f(x_i,u_i)$. We define $A_{\ell}(u_{1:\ell}\mid \xi) = \frac{\partial}{\partial x}f(x,u) \big{|}_{(x_{\ell}(u_{1:\ell}\mid \xi),u_{1:\ell})}$ and $B_{\ell}(u_{1:\ell} \mid \xi)$ analogously, but with $\frac{\partial}{\partial x}$ replaced by $\frac{\partial}{\partial u}$. We define the set of $\ell$-step linearizations $\Lins_{\ell}(B,\cX_0) := \bigcup_{u_{1:H} \in \cU_H(B),\xi \in \cX_0,0 \le h \le H-\ell}(A_{h:h+\ell}(u_{1:\ell}\mid \xi),B_{h:h+\ell}(u_{1:\ell}\mid \xi))$ as the set of $L$-step linearizations.

%We define the reachable set $\Xreach(B,\cX_0) := \bigcup_{u_{1:H} \in \cU_H(B),\xi \in \cX_0,h \in [H]}\{x_{h}(u_{1:h}\mid \xi)\}$. 
\begin{definition}[Stabilizability] We define
\begin{align}
J_{\ell}(h,B,\cX_0) := \sup_{\ell \in [h]}\sup_{(A_{1:\ell},B_{1:\ell}) \in \Lins_{\ell}(B,\cX_0)}\inf_{K_{1:\ell}} \left(\step\sum_{i=1}^{\ell} \trace(X_i^\top (I+K_{i}^\top K_i) X_i)\right) + \trace(X_{\ell+1}^\top X_{\ell+1}),
\end{align}
where $X_1 = \eye$, $X_{i+1} = (A_i+B_i K_i)X_i$. 
\end{definition}




An $\ell$-step randomized policy $\pi(\cdot \mid \zeta)$ is a mapping $\cX \to \cU^{\ell}$ parametrized by a seed $\seed \in \seedspace$; the set of all such policies is $\Pi_{[\ell]}$. We say $\pi$ is $B$-bounded if for all $x \in \cX$ and $\seed \in \seedspace$, $(u_1,\dots,u_{\ell}) = \pi(x \mid \seed)$ satisfies $\max_{i \in [\ell]}\|u_i\| \le B$. 

\begin{definition} We say that the dynamics $f$ are $(k,B,P)$. 
\end{definition}
