\begin{algorithm}
\caption{MCMC reverse diffusion sampling}
\label{alg:mcmc_reverse_sampling}
\begin{algorithmic}
\Require $\epsP(\cdot, t)$, $\beta_t$, $\alpha_t$, $\alphaBar_t$, $T$, $L$, $\langStep_t$, $\psi$ (remaining hyperparameters for MCMC method) % What time indices? Fill up with more hyper params
\State $\x_T \sim \normal{0, I}$
\For{$t = T, \dots, 1$}
    \Comment{Reverse diffusion}
    \State $\eps \sim \normal{0, I}$
    \State 
    $\x_{t-1} = \dfrac{1}{\sqrt{\alpha_t}} \left( \x_t - \dfrac{\beta_t}{\sqrt{1 - \alphaBar_t}} \epsilon_\theta(\x_t, t) \right) + \sqrt{\beta_t} \eps$ % Check correct eps scale
    \State $\x^{\tau=0} = \x_{t-1}$
    \If{$t > 1$} \Comment{MCMC-sampler at each $t-1$}
        \For{$\tau = 1, \dots, L$}
            \State 
            $\x^{\tau} = \text{MCMC-sampler}(\epsP, \x^{\tau-1}, \tau, t-1, \langStep_{t-1}, \psi)$
        \EndFor
        \State $\x_{t-1} = \x^{\tau = L}$
    \EndIf
\EndFor
\State \Return $\x_0$
\end{algorithmic}
\end{algorithm}
