\begin{algorithm}[t]
    \caption{Function for Candidate Patterns}
    \label{alg:candidate}
    \KwIn{A code table $CT$}
    \KwOut{A candidate code table $Cand$}
    \tcc{the explanation can be found in Sec. \ref{sec:candidates}}
    \For{{\bf each} $p_i \in CT$}{
        \For{{\bf each} $p_j \in CT$}{
            $Cand \leftarrow Cand \cup construct(p_i,~p_j)$\;
        }
    }
    \Return{$Cand$}
\end{algorithm}
