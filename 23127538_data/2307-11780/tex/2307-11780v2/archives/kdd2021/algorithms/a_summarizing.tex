\begin{algorithm}[t]
    \caption{Summarizing Algorithm}
    \label{alg:summarizing}
    \KwIn{The dataset $S$ and the standard code table $ST$}
    \KwOut{An approximate optimal code table $CT$}
    \tcc{the explanation can be found in Sec. \ref{sec:summarizing}}
    $CT \leftarrow ST$\;
    $Cand \leftarrow candidate(CT)$\;
    \For{{\bf each} $p \in Cand$ in Candidate Order}{
        \If{$L(S,~CT \cup \{p\}) < L(S,~CT)$}{
            $CT \leftarrow prune(S,~p,~CT) \cup \{p\}$\;
            $CT \leftarrow variations(S,~p,~CT)$\;
            $Cand \leftarrow candidate(CT)$\;
        }
    }
    \Return{$CT$}
\end{algorithm}
