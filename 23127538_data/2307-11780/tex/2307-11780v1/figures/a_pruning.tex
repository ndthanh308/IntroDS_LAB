\begin{algorithm}[t]
    \caption{Function for Pruning}
    \label{alg:pruning}
    \KwIn{The dataset $S$, a new pattern $p^*$, and a code table $CT$}
    \KwOut{A pruned code table $CT$}
    \tcc{the explanation can be found in Sec. \ref{sec:pruning}}
    $CT_p = CT \cup \{p^*\}$\;
    $PruneCand = \{p\in CT^*~|~usage(p~|~CT_p)<usage(p~|~CT)\}$\;
    \For{{\bf each} $p \in PruneCand$ in Prune Order}{
        \If{$L(S,~CT \setminus p) < L(S,~CT)$}{
            $CT \leftarrow CT \setminus p$\;
        }
    }
    \Return{$CT$}
\end{algorithm}
