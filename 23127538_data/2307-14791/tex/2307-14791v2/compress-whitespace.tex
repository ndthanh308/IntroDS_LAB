\makeatletter
\renewcommand{\@maketitle}{%
     \newpage
     \vspace{-9.75em} % <-- BEFORE TITLE
     \vbox to 2.5in{
        \vspace*{\fill}
        \vskip 2em
        \begin{center}%
            {\Large\bf \@title \par}%
            \vskip 0.375in minus 0.300in
            {\large\it
            \lineskip .5em
            \vspace{-0em} % <-- BETWEEN TITLE AND AUTHORS
            \begin{tabular}[t]{c}\@author
            \end{tabular}\par}%
        \end{center}%
        \par
        \vspace*{\fill}
        % \vskip 1.5em
        \vspace{-6em} % <-- AFTER AUTHOR
    }
}
\makeatother
