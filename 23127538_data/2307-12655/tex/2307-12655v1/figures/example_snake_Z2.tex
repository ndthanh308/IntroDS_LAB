\newcommand{\wang}[6]{
\draw [black,fill=#3] (#1,#2+1)  -- (#1+0.5,#2+0.5) -- (#1+1,#2+1) -- cycle;
\draw [black,fill=#4] (#1+1,#2+1)  -- (#1+0.5,#2+0.5) -- (#1+1,#2) -- cycle;
\draw [black,fill=#5] (#1,#2)  -- (#1+0.5,#2+0.5) -- (#1+1,#2) -- cycle;
\draw [black,fill=#6] (#1,#2)  -- (#1+0.5,#2+0.5) -- (#1,#2+1) -- cycle;
}
