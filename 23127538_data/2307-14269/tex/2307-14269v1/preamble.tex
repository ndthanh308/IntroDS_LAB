% Black/White markers
\pgfplotsset{
    stdmarks bw/.style={
        cycle list/Greys-5,
        cycle multiindex* list={
            mark list*\nextlist
            % Greys-F, Greys-G, Greys-H, Greys-I, Greys-J\nextlist
            Greys-5\nextlist
            draw=black
        },
        line join=round,
    },
    errormarks bw/.style={
        cycle list/Greys,
        cycle multiindex* list={
            mark list*\nextlist
            Greys-F, Greys-G, Greys-H, Greys-I, Greys-J\nextlist
            {},{},draw=black\nextlist
            linestyles\nextlist
        },
        mark options={solid,draw=black,scale=0.7,line width=0.2pt},
        line join=round,
    },
}
