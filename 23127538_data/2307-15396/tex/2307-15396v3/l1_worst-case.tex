\section{Tempered Overfiiting with respect to $\ell_1$-loss}
 \begin{theorem} 
 For the joint distribution $\dist$, we have that
 \[
 \lim_{n \rightarrow \infty} \Exp_S \sup_{\hat{f} \in \ridgeless(S)} [L_1(\hat{f})] \leq 50 \, L_1^*~,
 \]
 where $L_1^*=L_1(f^*)$ is $\ell_1$ the Bayes risk for the distribution $\dist$.
 \end{theorem}
 \begin{proof}
Again, we will have $n$ inteervals of length $\ell_1,\dots,\ell_n$ such that $\sum_{i=1}^{n} \ell_i=1$. For any $i\in \{2,\dots,n-1\}$ and any $x\in [x_{i-1},x_i)$, by Lemma \ref{?},
 \[
 |\hat{f}(x)-f^*(x)| \leq \max\left\{ |g_i(x)-f^*(x)|, \min\{|g_{i+1}(x)-f^*(x)|, |g_{i-1}(x)-f^*(x)|\}\right\}
 \]
 \begin{align*}
      |\hat{f}(x)-f^*(x)| &\leq \max\left\{ |g_i(x)-f^*(x)|, \min\{|g_{i+1}(x)-f^*(x)|, |g_{i-1}(x)-f^*(x)|\}\right\}\\
      & \leq |g_i(x)-f^*(x)|+ \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \\
      \int_{x_{i-1}}^{x_i} |\hat{f}(x)-f^*(x)| \, dx \, & \leq   \int_{x_{i-1}}^{x_i} |g_i(x)-f^*(x)| \, dx + \int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx \\
      \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} |\hat{f}(x)-f^*(x)| \, dx \, & \leq  \sum_{i=2}^{n-1} \int_{x_{i-1}}^{x_i} |g_i(x)-f^*(x)| \, dx + \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx \\
 \end{align*}
Moreover, for $i \in \{1,n\}$
 \[
 |\hat{f}(x)-f^*(x)| \leq |g_i(x)-f^*(x)|
 \]
 Therefore,
 $$ \int_{x_{i-1}}^{x_i}  |\hat{f}(x)-f^*(x)| \, dx \, \leq \int_{x_{i-1}}^{x_i} |g_i(x)-f^*(x)| \, dx$$
 Combining the above:
 \begin{align*}
     \Exp_{x \sim \unif([0,1])} [|\hat{f}(x)-f^*(x)|] &= \int_{0}^{1} |\hat{f}(x)-f^*(x)| \, dx \\
     &= \sum_{i=1}^{n} \int_{x_{i-1}}^{x_i} |\hat{f}(x)-f^*(x)| \, dx \\
     & \leq \sum_{i=1}^{n} \int_{x_{i-1}}^{x_i} |g_i(x)-f^*(x)| \, dx \\
     & \hspace{2mm} + \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx 
 \end{align*}
 Therefore, taking expectation with respect to $S$ and considering the limit as $n$ approaches $\infty$:
 \begin{align*}
    \lim_{n \rightarrow \infty} \Exp_S  \sup_{\hat{f} \in \ridgeless (S)} [L_1(\hat{f})] & \leq \lim_{n \rightarrow \infty} \Exp_S \sum_{i=1}^{n} \int_{x_{i-1}}^{x_i} |g_i(x)-f^*(x)| \, dx \\
     & \hspace{2mm} + \lim_{n \rightarrow \infty} \Exp_S \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx\\
     & \leq \lim_{n \rightarrow \infty}  \Exp_S \left[ \Exp_{x\sim \unif([0,1])} \left[ |\hat{f}_{\text{spline}}-f^*(x)| \right] \right] \\
     & \hspace{2mm} + \underbrace{\lim_{n \rightarrow \infty} \Exp_S \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx }_{:=T}\\
    & \leq  2 \cdot 4^{p} \cdot \sigma^p \cdot\sqrt{\frac{2^{p}}{\pi}} \Gamma\left( \frac{p+1}{2}\right) + T\\
 \end{align*}
 Therefore, it basically boils down to analyzing
 $$T=\lim_{n \rightarrow \infty} \Exp_S \sum_{i=2}^{n-1}\int_{x_{i-1}}^{x_i} \min\{|g_{i+1}(x)-f^*(x)|  , |g_{i-1}(x)-f^*(x)|\} \, dx$$
 \end{proof}
