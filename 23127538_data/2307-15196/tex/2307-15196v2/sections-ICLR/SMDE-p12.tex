
\subsection{Stochastic Differential Equation Results for SGDM}

\subsection{$\alpha\geq 1$}
When $\alpha=1$, the dynamics $\vx_k$ of SGDM  order-$1$ weakly approximated by $\bvx_{\eta k}$ where
\begin{align} 
\dd{\bvx_t} & = - \bvm_t \dd t\\
\dd{\bvm_t} & = - \lambda \bvm_t \dd t + \lambda\nabla\Loss(\bvx_t)\dd t + \lambda\Sigma^{1/2}(\bvx_t)\dd W_t
\end{align}
When $\alpha>1$, the dynamics $\vx_k$ has following facts:
\paragraph{Fact 1} The SGDM $\vx_k$ is an order-$(\alpha-1)$ weak approximation of 
\[\barvx_k = \vx_0 - \sum_{i=1}^k \eta_i m_0\] 
for steps $k=O(1/\eta)$.

\paragraph{Fact 2} The SGDM $\vx_k$ is an order-$(\alpha-1)/2$ weak approximation of $\bvx_t$. 
\begin{align} 
\dd{\bvx_t} & = - \bvm_t \dd t\\
\dd{\bvm_t} & = \lambda\nabla\Loss(\bvx_t)\dd t + \lambda\Sigma^{1/2}(\bvx_t)\dd W_t
\end{align}
with time scaling $t=\eta^{(\alpha+1)/2}k$ for steps $k=O(\eta^{-(\alpha+1)/2})$.

Consider a 1-dimensional example with $\Loss(x)=\frac{1}{2} ax^2$. The SGDM updates like
\begin{align*}
    \left(\begin{matrix} x_{k+1}\\ m_{k+1} \end{matrix}\right) = \left(\begin{matrix} 1-\eta(1-\beta)a & -\eta\beta\\ (1-\beta)a & \beta \end{matrix}\right)\left(\begin{matrix} x_{k}\\ m_{k} \end{matrix}\right) + \left(\begin{matrix}-\eta\\ 1 \end{matrix}\right) (1-\beta)\sigma v_k
\end{align*}
Let $\kappa=\frac{1}{2}(1+\beta-a\eta(1-\beta))$.
The matrix have eigenvalues
\[\lambda = \kappa\pm\sqrt{\kappa^2-\beta}.\]
$\kappa\sim 1-O(\eta^\alpha)$, $\kappa^2-\beta\sim -O(\eta^{1+\alpha})$
