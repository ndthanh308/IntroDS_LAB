





We recall the definition of $T_1$-$T_9$ listed as below.  We will continue by proving tighter bounds for some of these terms with respect to $v^*$.

\begin{align}
& T_1= \sum_{k=1}^K \sum_{h=1}^H \left( \hat{P}^k_{s_h^k,a_h^k,h}-P_{s_h^k,a_h^k,h} \right)V_{h+1}^k ;\nonumber
\\ &  T_2 = \sum_{k=1}^K \sum_{h=1}^H b_h^k(s_h^k,a_h^k);\nonumber
\\ & T_3 =\sum_{k=1}^K \sum_{h=1}^H (P_{s_h^k,a_h^k,h}-\textbf{1}_{s_{h+1}^k})V_{h+1}^k;\nonumber
\\ & T_4 = \sum_{k=1}^K \left( \sum_{h=1}^H \hat{r}_h^k(s_h^k,a_h^k)-V_{1}^{\pi^k}(s_1^k)\right);\nonumber
\\ & T_5  = \sum_{k=1}^K \sum_{h=1}^H \mathbb{V}(\hat{P}^k_{s_h^k,a_h^k,h},V_{h+1}^k);\nonumber
\\ & T_6 = \sum_{k=1}^K \sum_{h=1}^H \mathbb{V}(P_{s_h^k,a_h^k,h},V_{h+1}^k);\nonumber
\\ & T_7  = \sum_{k=1}^K \sum_{h=1}^H \left( \hat{P}^k_{s_h^k,a_h^k,h} - P_{s_h^k,a_h^k,h} \right) (V_{h+1}^k)^2;\nonumber
\\ & T_8 =\sum_{k=1}^K \sum_{h=1}^H (P_{s_h^k,a_h^k,h}-\textbf{1}_{s_{h+1}^k})(V_{h+1}^k)^2;\nonumber
\\ & T_9  = \sum_{k=1}^K\sum_{h=1}^H \max\left\{ (\hat{P}^k_{s_h^k,a_h^k,h}-P_{s_h^k,a_h^k,h})V_{h+1}^k, 0  \right\}.\nonumber
\end{align}

Also recall \eqref{eq:obt2}-\eqref{eq:obt9}. We will provide refined analysis for the bound of $T_2$, $T_4$, $T_5$ and $T_6$, and leave other bounds invariant.


\paragraph{Bound of $T_2$}
Recall that in \eqref{eq:boundt2o}, we show that
\begin{align}
T_2 &\leq \frac{460}{9}\sqrt{2SAH\log_2(K)\log(\frac{1}{\delta})T_5} \nonumber
\\ & \qquad \qquad + 4\sqrt{SAH^2\log_2(K)\log(\frac{1}{\delta})}\sqrt{\sum_{k=1}^K\sum_{h=1}^H \hat{r}_h^k(s_h^k,a_h^k)}+ \frac{1088}{9}SAH^2\log_2(K)\log(\frac{1}{\delta}).\nonumber
\end{align}

By definition of $T_4$, and the fact that $\sum_{k=1}^K V_1^*(s_1^k)\leq 3Kv^* + H\log(\frac{1}{\delta})$ with probability $1-\delta$ (Lemma~\ref{lemma:con}),
 it then holds that
 \begin{align}
T_2&\leq \frac{460}{9}\sqrt{2SAH\log_2(K)\log(\frac{1}{\delta})T_5} \nonumber
\\ & \qquad \qquad + 4\sqrt{SAH^2\log_2(K)\log(\frac{1}{\delta})}\sqrt{T_4 + 3Kv^*}+ 130SAH^2\log_2(K)\log(\frac{1}{\delta}).\label{eq:nbft2}
 \end{align}


\paragraph{Bound of $T_4$}
Note that
\begin{align}
 T_4 & = \sum_{k=1}^K \left( \sum_{h=1}^H \hat{r}_h^k(s_h^k,a_h^k)-V_{1}^{\pi^k}(s_1^k)\right) \nonumber
 \\ & = \sum_{k=1}^K \left( \sum_{h=1}^H \hat{r}_h^k(s_h^k,a_h^k)-r_{h}(s_h^k,a_h^k)\right) + \sum_{k=1}^K \left( \sum_{h=1}^H r_h(s_h^k,a_h^k) - V_{1}^{\pi^k}(s_1^k) \right).
 \end{align}

Let $\check{T}_1 =\sum_{k=1}^K \left( \sum_{h=1}^H \hat{r}_h^k(s_h^k,a_h^k)-r_{h}(s_h^k,a_h^k)\right)  $ and $\check{T}_2 =\sum_{k=1}^K \left( \sum_{h=1}^H r_h(s_h^k,a_h^k) - V_{1}^{\pi^k}(s_1^k) \right) $
By Lemma~\ref{lemma:bdempr} and Lemma~\ref{lemma:doubling}, with probability $1-2SAHK\delta$,  
\begin{align}
 \check{T}_1 & \leq 4\sqrt{2SAH^2\log_2(K)}\cdot \sqrt{\sum_{k=1}^K\sum_{h=1}^H r_h(s_h^k,a_h^k)} + 52SAH^2\log_2(K)\log(\frac{1}{\delta}) \nonumber
  \\ & \leq 4\sqrt{2SAH^2\log_2(K)}\cdot \sqrt{\check{T}_2 + 3Kv^*} + 60SAH^2\log_2(K)\log(\frac{1}{\delta}).\label{eq:ct1}
\end{align}

On the other hand, by Lemma~\ref{lemma:self-norm}, with probability $1-2SAHK\delta$,
\begin{align}
\check{T}_2 & \leq 2\sqrt{2\sum_{k=1}^K \mathbb{E}_{\pi^k,s_1\sim \mu_1}\left[\left(\sum_{h=1}^H r_h(s_h,a_h) \right)^2 \right]\log(\frac{1}{\delta})}+3H^2\log(\frac{1}{\delta}) \nonumber
\\ & 2\sqrt{2H\sum_{k=1}^K \mathbb{E}_{\pi^k,s_1\sim \mu_1}\left[\sum_{h=1}^H r_h(s_h,a_h)  \right]\log(\frac{1}{\delta})}+3H\log(\frac{1}{\delta}) \nonumber
\\ & \leq 2\sqrt{2KHv^*\log(\frac{1}{\delta})}+3H\log(\frac{1}{\delta}) \label{eq:ct1.5}
\\ & \leq 2Kv^* + 5H\log(\frac{1}{\delta}).\label{eq:ct2}
\end{align}

Combining \eqref{eq:ct1}, \eqref{eq:ct1.5} with \eqref{eq:ct2}, with probability $1-4SAHK\delta$,
\begin{align}
& \check{T}_1\leq     12\sqrt{SAH^2Kv^*\log_2(K)\log(\frac{1}{\delta})}  + 80SAH^2\log_2(K)\log(\frac{1}{\delta})\nonumber
\\ & \check{T}_2 \leq  2\sqrt{2KHv^*\log(\frac{1}{\delta})}+3H\log(\frac{1}{\delta}) .\nonumber
\end{align}

As a result, we have that
\begin{align}
T_4 \leq 14sqrt{SAH^2Kv^*\log_2(K)\log(\frac{1}{\delta})}  + 83SAH^2\log_2(K)\log(\frac{1}{\delta}).\label{eq:fnbt4}
\end{align}

\paragraph{Bound of $T_5$}

Recall that in \eqref{eq:boundt5}, we show that
\begin{align}
T_5  & \leq \sum_{k=1}^K \sum_{h=1}^H   (\hat{P}^k_{s_h^k,a_h^k,h} -P_{s_h^k,a_h^k,h})(V^k_{h+1})^2   + \sum_{k=1}^K \sum_{h=1}^H (P_{s_h^k,a_h^k,h} - \textbf{1}_{s_{h+1}^k} )  (V_{h+1}^k)^2  \nonumber
\\ & \qquad \qquad\qquad \qquad \qquad \qquad   +2H \sum_{k=1}^K \sum_{h=1}^H b_h^k(s_h^k,a_h^k)+2H\sum_{k=1}^K \sum_{h=1}^H r_h(s_h^k,a_h^k).\nonumber
\end{align}

With \eqref{eq:ct2}, with probability $1-4SAHK\delta$,
\begin{align}
T_5\leq  T_7 + T_8+2HT_2 + 4HKv^* + 10H^2\log(\frac{1}{\delta}).\label{eq:fnbt5}
\end{align}


\paragraph{Bound of $T_6$}
Recall \eqref{eq:boundt6}
\begin{align}
T_6 & \leq 2\sqrt{8T_6\log(\frac{1}{\delta})} + 3H^2\log(\frac{1}{\delta})+2H \sum_{k=1}^K\sum_{h=1}^H \max\{V_h^k(s_h^k)-P_{s_h^k,a_h^k,h}V_{h+1}^k,0 \} \nonumber
\\ & \leq  2\sqrt{8T_6\log(\frac{1}{\delta})} + 3H^2\log(\frac{1}{\delta})+2HT_2 + 2HT_9 + 2H\sum_{k=1}^K\sum_{h=1}^H r_h(s_h^k,a_h^k)\nonumber
\\ & \leq \leq 2\sqrt{8T_6\log(\frac{1}{\delta})} + 3H^2\log(\frac{1}{\delta})+2H \sum_{k=1}^K\sum_{h=1}^H \max\{V_h^k(s_h^k)-P_{s_h^k,a_h^k,h}V_{h+1}^k,0 \} \nonumber
\\ & \leq  2\sqrt{8T_6\log(\frac{1}{\delta})} + 2HT_9 + 4HKv^*+ 6H^2\log(\frac{1}{\delta})+2HT_2 .\label{eq:fnbt6}
\end{align}

\paragraph{Putting all together}

Assume that $K\geq \frac{BSAH^2}{v^*}$. Then $\sqrt{BSAH^2Kv^*}\geq BSAH^2$. 
Solving \eqref{eq:nbft2},\eqref{eq:boundt3},\eqref{eq:fnbt4},\eqref{eq:fnbt5},\eqref{eq:nbt6},\eqref{eq:boundt8},\eqref{eq:boundt1},\eqref{eq:boundt7} and \eqref{eq:boundt9}, we have that, with probability $1-100SAH^2K\delta$, $T_6 = O(BHKv^*+BSAH^3)$, $T_1 = O(\sqrt{BSAH^2Kv^*})$, $T_7,T_8 = O(\sqrt{BSAH^4Kv^*})$, $T_5 = O(BHKv^*)$, $T_2 = O(\sqrt{BSAH^2Kv^*})$ and $T_3 = O(\sqrt{BHKv^*})$. We then conclude that the total regret is bounded by $O(\sqrt{BSAH^2Kv^*})$. In the case $K\leq \frac{BSAH^2}{v^*}$, the regret bound is trivially $O(Kv^*)$. The proof is completed by replacing $\delta$ with $\frac{\delta}{100SAH^2K}$.
